-- Create Tables
DROP TABLE IF EXISTS Books;
CREATE TABLE Books (
    Book_ID SERIAL PRIMARY KEY,
    Title VARCHAR(100),
    Author VARCHAR(100),
    Genre VARCHAR(50),
    Published_Year INT,
    Price NUMERIC(10, 2),
    Stock INT
);

DROP TABLE IF EXISTS customers;
CREATE TABLE Customers (
    Customer_ID SERIAL PRIMARY KEY,
    Name VARCHAR(100),
    Email VARCHAR(100),
    Phone VARCHAR(15),
    City VARCHAR(50),
    Country VARCHAR(150)
);
DROP TABLE IF EXISTS orders;
CREATE TABLE Orders (
    Order_ID SERIAL PRIMARY KEY,
    Customer_ID INT REFERENCES Customers(Customer_ID),
    Book_ID INT REFERENCES Books(Book_ID),
    Order_Date DATE,
    Quantity INT,
    Total_Amount NUMERIC(10, 2)
);

--10 Calcular o valor médio gasto por cada cliente
SELECT Customer_ID, AVG(Total_Amount) AS Media_Gasta
FROM Orders
GROUP BY Customer_ID;

-- 11.Recuperar o total de livros vendidos por gênero
SELECT b.Genre, SUM(o.Quantity) AS Total_Vendido
FROM Orders o
JOIN Books b ON o.Book_ID = b.Book_ID
GROUP BY b.Genre;

-- 12.Calcular o preço médio dos livros do gênero "Fantasy"
SELECT 
	to_char(AVG(Price),'FM999G999G999') AS Preco_Medio
FROM Books
WHERE Genre = 'Fantasy';

-- 13.Listar os gêneros com maior receita de vendas
SELECT b.Genre, SUM(o.Total_Amount) AS Receita
FROM Orders o
JOIN Books b ON o.Book_ID = b.Book_ID
GROUP BY b.Genre
ORDER BY Receita DESC;
